Build Your AI Chatbot

chatbot-flow.js 
// Create a new chatflow
const chatflow = new ChatFlow();

// Add OpenAI node
chatflow.addNode({
    type: "llm",
    model: "gpt-3.5-turbo",
    temperature: 0.7
});

// Add conversation memory
chatflow.addNode({
    type: "memory",
    chatHistory: true
});

// Connect to API endpoint
chatflow.deploy({
    endpoint: "/api/chat"
});
